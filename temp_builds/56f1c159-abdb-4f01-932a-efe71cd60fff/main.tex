% Simple framed box using standard LaTeX
\newenvironment{qbox}{%
  \par\noindent
  \fbox{\begin{minipage}{\dimexpr\linewidth-2\fboxsep-2\fboxrule\relax}}
}{%
  \end{minipage}}
  \vspace{0.5em}
}
