% Simple framed box using standard LaTeX
\newsavebox{\myqbox}

  \begin{lrbox}{\myqbox}%
  \begin{minipage}{\dimexpr\linewidth-2\fboxsep-2\fboxrule\relax}
}{%
  \end{minipage}%
  \end{lrbox}%
  \par\noindent\fbox{\usebox{\myqbox}}%
  \vspace{0.5em}
}
